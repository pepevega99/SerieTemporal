import pandas as pd
import matplotlib.pyplot as plt
import statsmodels.api as sm
In[28]:
# Realizar una descomposicin de la serie temporal
decomposition = sm.tsa.seasonal_decompose(viajeros_clean['Visitantes totales'], model='additive')

# Graficar los componentes descompuestos
plt.figure(figsize=(14, 7))
plt.subplot(411)
plt.plot(decomposition.observed, label='Original')
plt.legend(loc='best')
plt.subplot(412)
plt.plot(decomposition.trend, label='Trend')
plt.legend(loc='best')
plt.subplot(413)
plt.plot(decomposition.seasonal, label='Seasonality')
plt.legend(loc='best')
plt.subplot(414)
plt.plot(decomposition.resid, label='Residuals')
plt.legend(loc='best')
plt.tight_layout()
plt.show()
